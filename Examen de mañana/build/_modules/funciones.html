[documentos]def cuadrados (lista):
    """
    Funcion que hace el cuadrado de los elementos de una lista
    :param lista: lista de numeros
    :return: lista de cuadrados

    >>> lista = [1,2,3,4]
    >>> cuadrados(lista)
    [1, 4, 9, 16]
    """
    return [n ** 2 for n in lista]
